// Add the following blocks of code to any part.cfg to bestow capabilities needed for this mod
// If you want, you can add ALL of them to one part, eg the launchpad!

// Add this if you need to cheat power
MODULE
{
	name = ModuleGenerator
	isAlwaysActive = true
	OUTPUT_RESOURCE
	{
	   name = ElectricCharge
	   rate = 300
	}	
}

RESOURCE
{
	name = ElectricCharge
	amount = 500
	maxAmount = 500
}

// Kethane compatibility
// FOR KETHANE 0.6 AND GREATER ONLY

// Ability to detect Ore
MODULE
{
    name = KethaneDetector
    DetectingPeriod = 1.5
    DetectingHeight = 250000
    PowerConsumption = 0.8
    Resource
    {
        Name = Ore
    }
}

// Can extract Ore (A "Drill") - does not need to touch ground
// If using the kethane drill part, just replace the KethaneExtractor block with this one
MODULE
{
    name = KethaneExtractor
    PowerConsumption = 24
    Resource
    {
        Name = Ore
        Rate = 5
    }
}

// Ore extractor part may also need this
MODULE
{
    name = KethaneExtractorAnimatorLanded 
}

// Convert Ore to Metal
MODULE
{
    name = KethaneConverter
    SourceResource = Ore
    TargetResource = Metal
    ConversionEfficiency = 1
    SourceConsumption = 3
    PowerConsumption = 10
}

// Convert Metal to RocketParts
MODULE
{
    name = KethaneConverter
    SourceResource = Metal
    TargetResource = RocketParts
    ConversionEfficiency = 1
    SourceConsumption = 3
    PowerConsumption = 10
}

// Resource Storage
RESOURCE
{
	name = Metal
	amount = 0
	maxAmount = 10
}

RESOURCE
{
	name = Ore
	amount = 5
	maxAmount = 100
}

RESOURCE
{
	name = RocketParts
	amount = 0
	maxAmount = 200
}
